cd test   - переходим в папку test (если это папка и она существует)
cd ..     - переходим родительскую папку
cd .      - переход в эту же папку (мы никуда не перейдем)
cd        - переход в домашнюю папку(~)
cd ~      - переход в домашнюю папку(~)
pwd       - показывает абсолютный путь до текущей папки
mkdir test   - создание новой папки 
mkdir test/new_dir   - сщздание новой папки new_dir в уже существующей папке test
touch test.txt       - создание файла test.txt
rm test.txt          - удаление файла test.txt
rm -rf test          -удаление папки test со всем ее содержимым
sudo rm -rf /        -НИКОГДА НЕ ИСПОЛЬЗОВАТЬ
cat test.txt         - это просмотр содержимого файла test.txt
ls test              -просмотр всех фалов и папок внутри папки test
ls -a                -просмотр всех фалов и папок внутри папки test включая скрытые
--help      -к любой команде можно дописать и выйдет описание как исползовать эту команду
ls -l       -просмотр всех фалов и папок с более подробной инфо

nano test.txt -открыть файл в редакторе nano что бы сохранить (Ctrl s) что бы выйти(Ctrl x)
